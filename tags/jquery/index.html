end of jquery resize event

var rtime = new Date(1, 1, 2000, 12,00,00);

var timeout = false;

var delta = 200;

$(window).resize(function() {

rtime = new Date();

if (timeout === false) {

timeout = true;

setTimeout(resizeend, delta);

}

});

function resizeend() {

if (new Date() – rtime < delta) {

setTimeout(resizeend, delta);

} else {

timeout = false;

alert(‘Done resizing’);

}

}
