Setup

npm install --save-dev eslint eslint-plugin-react eslint-plugin-react-native eslint-config-airbnb

.eslintrc

eslint —init

edit

加入 extends airbnb 和 react plugin 就可以了

/* .eslintrc.js */
module.exports = {
    "extends": "airbnb",
    "plugins": [
        "react",
        "eslint-plugin-react-native"
    ],
    "parser": "babel-eslint",
    "rules": {
      "strict": 0
    },
};

nvm

profile

export NVM_DIR="$HOME/.nvm"
[ -s "$NVM_DIR/nvm.sh" ] && . "$NVM_DIR/nvm.sh" # This loads nvm

source ~/.zshrc
