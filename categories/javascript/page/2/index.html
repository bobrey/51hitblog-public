reduce

 

[0, 1, 2, 3, 4, 5].reduce((previous, current) => {
    return current + previous;
});


  0 + 1 = 1
  1 + 2 = 3
  3 + 3 = 6
  6 + 4 = 10
  10 + 5 = 15
