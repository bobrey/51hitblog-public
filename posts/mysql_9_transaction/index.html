#有时，当您从父表删除数据，可能想知道哪些表受MySQL DELETE CASCADE参照操作影响。那么可以执行以下 SQL 查询 - 
USE information_schema;
SELECT
	table_name 
FROM
	referential_constraints 
WHERE
	constraint_schema = 'database_name' 
	AND referenced_table_name = 'parent_table' 
	AND delete_rule = 'CASCADE'

#事务
-- 开始事务
start transaction;

SELECT @order_id:=max(order_id) FROM studymysql.order;
set @order_id=@order_id+1;
 
-- insert a new order for customer 1020
INSERT INTO studymysql.order(order_id, order_number,order_date,shipped_date,total_money,status,comment,customer_id)
values(@order_id, 609095688,'2016-09-10 22:21','2016-09-11 12:20',1198,'0','comment ... ',1020);

-- 插入两个商品信息
insert into studymysql.order_detail(order_id,product_id,buy_number,price,total_money)
values(@order_id, 47, 1, 999, 999), (@order_id, 51, 1, 199, 199);

-- 提交事务
COMMIT;      
 
-- 查询上面执行的结果，是否有新插入的数据？
SELECT *  FROM studymysql.order a
inner join studymysql.order_detail b on a.order_id=b.order_id
where a.order_id=@order_id;
