CREATE TABLE studymysql.contacts (
    id INT AUTO_INCREMENT PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    fullname varchar(101) GENERATED ALWAYS AS (concat(last_name,' ',first_name ))    
);

ALTER TABLE studymysql.product  ADD COLUMN stock_value DOUBLE  GENERATED ALWAYS AS (price*quantity_stock) STORED;



-- REGEX：基于正则表达式搜索; 以下语句仅匹配以大写“H”开头的产品名称。
SELECT  product_name FROM product WHERE product_name REGEXP BINARY '^H';
