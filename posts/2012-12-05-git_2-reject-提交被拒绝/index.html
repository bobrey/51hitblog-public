push 被reject

不能简单的git pull，然后git push；否则会造成版本混乱

git fetch origin

git rebase origin/master

若还有conflict，则修改冲突文件，然后

git add .

git rebase –continue

git push

———————————————————–

vim ~/.git/refs/remote/origin/master

vim ~/.gitconfig

[user]

name = xxx

email = xxxxxx@xxxx

[core]

editor = vim

[alias]

ci = commit -a -v

co = checkout

st = status

br = branch

throw = reset–hard HEAD

throwh = reset –hard HEAD^

[color]

ui = true

[push]

default = current
