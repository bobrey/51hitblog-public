    7.使用Migration和Seeder

    #php artisan make:migration create_articles_table
#vim learnlaravel5/database/migrations
  public function up(){
      Schema::create('articles',function(Blueprint $table){
          $table->increments('id');
          $table->string('title');
          $table->text('body')->nullable();
          $table->integer('user_id');
          $table->timestamps();
      })
  }
  #php artisan migrate
     
  //使用seeder
  #php artisan make:seeder ArticleSeeder
  # vim /learnlaravel5/database/seeds/ArticleSeeder.php
    public function run(){
        DB::table('articles')->delete();
        for($i=0;$i<10;$i++){
            \App\Article::create([
                'title'    =>'Title'.$i,
                'body'     =>'Body'.$i,
                'user_id'  =>1,
            ])
        }
    }
       
    由于 database 目录没有像 app 目录那样被 composer 注册为 psr-4 自动加载，采用的是 psr-0 classmap 方式，所以我们还需要运行以下命令把 ArticleSeeder.php 加入自动加载系统，避免找不到类的错误
    #composer dump-autoload
