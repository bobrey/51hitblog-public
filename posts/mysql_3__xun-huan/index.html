-- WHILE循环
DELIMITER $$
 DROP PROCEDURE IF EXISTS mysql_while_loop$$
 CREATE PROCEDURE mysql_while_loop()
 BEGIN
	DECLARE x  INT;
	DECLARE str  VARCHAR(255);

	SET x = 1;
	SET str =  '';

	WHILE x  <= 10 DO
		SET  str = CONCAT(str,x,',');
		SET  x = x + 1; 
	END WHILE;

 SELECT str;
 END$$
DELIMITER ;

CALL mysql_while_loop();






-- --REPEAT循环
DELIMITER $$
 DROP PROCEDURE IF EXISTS mysql_repeat_loop$$
 CREATE PROCEDURE mysql_repeat_loop()
 BEGIN
	DECLARE x INT;
	DECLARE str VARCHAR(255);

	SET x = 1;
  SET str =  '';

 REPEAT
	SET  str = CONCAT(str,x,',');
	SET  x = x + 1; 
  UNTIL x  > 10
END REPEAT;

SELECT str;

 END$$
 DELIMITER ;








-- LOOP，LEAVE和ITERATIVE语句

DELIMITER $$
DROP PROCEDURE IF EXISTS test_mysql_loop$$
CREATE PROCEDURE test_mysql_loop()
BEGIN
 DECLARE x  INT(10);
 DECLARE str VARCHAR(255);

 SET x = 1;
 SET str =  '';

 loop_label: LOOP
     IF  x > 10 THEN 
				LEAVE  loop_label;
     END  IF;

     SET  x = x + 1;
     IF  (x mod 2) THEN
     ITERATE  
        loop_label;
     ELSE
        SET  str = CONCAT(str,x,',');
     END  IF;
 END LOOP;   

 SELECT str;

 END$$
 DELIMITER ;
