Ntfs挂载

mac挂载ntfs硬盘

  diskutil list
    mkdir ~/Desktop/Windows	
    sudo umount /dev/disk2s4
    sudo mount_ntfs -o rw,nobrowse /dev/disk2s4 ~/Desktop/Windows

debian挂载ntfs硬盘

    apt-get install  fuse-utils ntfs-3g
    fdisk -l
    mkdir /mnt/<DIR>
    mount -t ntfs-3g /dev/<sdXX>  /mnt/<DIR>
