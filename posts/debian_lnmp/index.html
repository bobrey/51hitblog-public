Debian_lnmp


支持PHP7.2、MySQL5.7

1.nginx
  apt-get install nginx
  apt-get install restart


2.mysql
  apt-get install mysql-server mysql-client
  mysql -u root -p
    use mysql;
    update mysql.user set authentication_string=PASSWORD('gy!'), plugin='mysql_native_password' where user='root';
    flush privileges;
  service mysql restart 
  mysql -u root -p




3.php7.2.5
    wget -q https://packages.sury.org/php/apt.gpg -O- | sudo apt-key add -
    echo "deb https://packages.sury.org/php/ stretch main" | sudo tee /etc/apt/sources.list.d/php.list
    apt-get install php7.2 php7.2-fpm php7.2-mysql php7.2-memcached
    
    vim /etc/nginx/sites-enables/default
      location ~ \.php$ {
        include snippets/fastcgi-php.conf;
        fastcgi_pass unix:/run/php/php7.2-fpm.sock;
      }
      
    service nginx restart  

    vim /etc/php/7.2/fpm/pool.d/www.conf
      listen = /run/php/php7.2-fpm.sock
    service php7.2-fpm restar
