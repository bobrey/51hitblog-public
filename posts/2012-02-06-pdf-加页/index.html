UIGraphicsBeginPDFContextToFile(pdfFileName, CGRectZero, nil);

for (int j = 1; j <= pageCount; j++) {

NSMutableArray *markerList = [[AppDocument sharedDocument] loadWhiteboardFromFilePath:[NSString stringWithFormat:@”%@/%d”,path,j]];

DLog(@”%@”,[NSString stringWithFormat:@”%@/%d”,path,j]);

//            UIGraphicsBeginPDFPageWithInfo(CGRectMake(0, 0, 1024, 768),nil);

//            UIGraphicsBeginPDFPageWithInfo(CGRectMake(0, 0, 842, 595),nil);

UIGraphicsBeginPDFPageWithInfo(CGRectMake(0, 0, 920, 660),nil);

CGContextRef context = UIGraphicsGetCurrentContext();

//CGContextScaleCTM(context, 1.0, -1.0);

if (markerList != nil && [markerList count] > 0) {

for (int i=0; i<[markerList count]; i++) {

LineMessage *thisArray = [markerList objectAtIndex:i];

if ([[thisArray linePoints] count] > 2) {

float size = [[thisArray lineSize] floatValue];

float r = [[[thisArray colorRGB] objectAtIndex:0] floatValue];

float g = [[[thisArray colorRGB] objectAtIndex:1] floatValue];

float b = [[[thisArray colorRGB] objectAtIndex:2] floatValue];

UIColor *color=[UIColor colorWithRed:r

green:g

blue:b

alpha:0.5];

//float size=4;

//UIColor *color=[UIColor blueColor];

CGContextSetLineWidth(context, size);

CGContextSetStrokeColorWithColor(context,color.CGColor);

CGContextSetFillColorWithColor(context,color.CGColor);

float thisX = [[[thisArray linePoints] objectAtIndex:0] floatValue];

float thisY = [[[thisArray linePoints] objectAtIndex:1] floatValue];

CGContextBeginPath(context);

CGContextMoveToPoint(context, thisX, thisY);

for (int j=2; j<[[thisArray linePoints] count]; j+=2) {

thisX = [[[thisArray linePoints] objectAtIndex:j] floatValue];

thisY = [[[thisArray linePoints] objectAtIndex:j+1] floatValue];

CGContextAddLineToPoint(context, thisX, thisY);

}

CGContextStrokePath(context);

}

}

}

}

UIGraphicsEndPDFContext();
