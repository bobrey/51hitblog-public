# 临时表

CREATE TEMPORARY TABLE top10customer
SELECT p.customer_id,
       c.customer_name,
       FORMAT(SUM(p.total_money), 2) total
FROM studymysql.payment p
INNER JOIN studymysql.customer c ON c.customer_id=p.customer_id
GROUP BY p.customer_id
ORDER BY total DESC
LIMIT 10;

#L检查约束模拟
CREATE TABLE IF NOT EXISTS studymysql.part (
    part_no VARCHAR(18) PRIMARY KEY,
    description VARCHAR(40),
    cost DECIMAL(10 , 2 ) NOT NULL,
    price DECIMAL(10,2) NOT NULL
);


#创建一个存储过程来检查 cost 和 price 列的值
DELIMITER $$
CREATE PROCEDURE `check_parts`(IN cost DECIMAL(10,2), IN price DECIMAL(10,2))
BEGIN
    IF cost < 0 THEN
        SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'check constraint on part.cost failed';
    END IF;

    IF price < 0 THEN
			SIGNAL SQLSTATE '45001'
				SET MESSAGE_TEXT = 'check constraint on part.price failed';
    END IF;

    IF price < cost THEN
			SIGNAL SQLSTATE '45002'
            SET MESSAGE_TEXT = 'check constraint on part.price & part.cost failed';
    END IF;

END$$
DELIMITER ;


#创建 BEFORE INSERT 和 BEFORE UPDATE 触发器。触发器中，调用 check_parts（）存储过程
-- before insert
DELIMITER $$
CREATE TRIGGER `parts_before_insert` BEFORE INSERT ON `part`
FOR EACH ROW
BEGIN
    CALL check_parts(new.cost,new.price);
END$$   
DELIMITER ; 

-- before update
DELIMITER $$
CREATE TRIGGER `parts_before_update` BEFORE UPDATE ON `part`
FOR EACH ROW
BEGIN
    CALL check_parts(new.cost,new.price);
END$$   
DELIMITER ;
