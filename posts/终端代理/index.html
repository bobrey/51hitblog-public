终端走代理

vim ~/.zshrc

fuction proxy()
{
  export http_proxy=http://127.0.0.1:1087
  export https_proxy=http://127.0.0.1:1087 
  export ALL_PROXY=socks5://127.0.0.1:1080
}
fuction noproxy()
{
  unset http_proxy
  unset https_proxy
  unset ALL_PROXY
}

function  proxyWifi()
{
  networksetup -setsocksfirewallproxy 'Wi-Fi' 127.0.0.1 1080
}
function noproxyWifi()
{
   networksetup -setsocksfirewallproxystate 'Wi-Fi' off
}

alias ip="curl ip.cn && curl cip.cc && curl ip.gs"
alias v2ray="nohup /usr/local/v2ray-macos/v2ray >> /tmp/v2ray.log 2>&1 & "
