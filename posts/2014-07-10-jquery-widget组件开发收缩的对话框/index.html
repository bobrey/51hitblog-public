jquery widget组件开发,

可以收放dialog,自定义对话框；

$.widget( “custom.superDialog”, $.ui.dialog, {

blue: function() {

this.element.css( “background”, “lightblue” );

},

_setOption:function ( key, value ){

this._super( key, value );

this._superApply ( arguments );

},

_init:function(){

var dialog_id = this.uiDialogTitlebar.next().attr(‘id’);

this.uiDialogTitlebar.append(‘[‘ +

‘]1{#' + dialog_id +'-minbutton.ui-dialog-titlebar-minimize.ui-corner-all}‘);

$(‘#’ + dialog_id + ‘-minbutton’).hover(function () {

$(this).addClass(‘ui-state-hover’);

},function () {

$(this).removeClass(‘ui-state-hover’);

}).click(function () {

$(‘#’ + dialog_id).toggle();

var $icon_span = $(this).find(“span”);

$icon_span.toggleClass(function(){

if($icon_span.hasClass(“ui-icon-minusthick”) ){

$icon_span.removeClass();

return “ui-icon ui-icon-plusthick”;

}else{

$icon_span.removeClass();

return “ui-icon ui-icon-minusthick”;

}

}); //end change class

});

this._superApply(this, arguments );

} //end init

} );

}
