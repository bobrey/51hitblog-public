Harbor

安装

wget https://storage.googleapis.com/harbor-releases/release-1.8.0/harbor-offline-installer-v1.8.1.tgz
tar zxvf 
cd harbor 

vim harbor
	hostname: 192.168.50.156
	port: 5000
	harbor_admin_password= 123456
	data_volume: /usr/src/harbor/data
	location: /var/log/harbor

vim docker-compose.yml
	ports:
      #需要改这里
      - 5000:80
      - 443:443
      - 4443:4443

./install
docker-compose start/stop

使用

创建wyx用户
创建test项目
为test项目添加wyx往上，角色为项目管理员

docker login 192.168.50.156:5000
Username: wyx
Password



//由于docker registray使用的是https，而目前Harborsudsy的http
//查找docker.service目录; [增加 --insecure-registry 192.168.50.156:5000]
find / -name docker.service -type f
vim /usr/lib/systemd/system/docker.service
    ExecStart=/usr/bin/dockerd --insecure-registry 192.168.50.156:5000 

# 重新加载配置、重启docker
systemctl daemon-reload
systemctl restart docker
 
 //再次登录
 docker login 192.168.50.156:5000

推拉镜像

# 拉取一个测试镜像
docker pull daocloud.io/daocloud/phpmyadmin

# 标记本地镜像, 将其归入某一仓库(harbor.phpdev.com)
docker tag daocloud.io/daocloud/phpmyadmin:latest 192.168.50.156:5000/test/phpmyadmin:v1

# 将本地镜像推送到镜像仓库(需先登录镜像仓库)
docker push 192.168.50.156:5000/test/phpmyadmin:v1


# 删除原来镜像
docker rmi daocloud.io/daocloud/phpmyadmin
docker rmi 192.168.50.156:5000/test/phpmyadmin:v1
# 拉取
docker pull 192.168.50.156:5000/test/phpmyadmin:v1
