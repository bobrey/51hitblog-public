ansible介绍

1. 安装ansible
  $ sudo apt-get update
  $ sudo apt-get install software-properties-common
  $ sudo apt-add-repository --yes --update ppa:ansible/ansible
  $ sudo apt-get install ansible
  ansible --version
  ssh-keygen
  ssh-copy-id guo@172.16.62.151
  ssh-copy-id guo@172.16.62.152
  ssh-keyscan 172.16.62.151 >> ~/.ssh/known_hosts
  ssh-keyscan 172.16.62.152 >> ~/.ssh/known_hosts
  验证
    ssh guo@172.16.62.151
    ssh guo@172.16.62.152

2.
  配置
  vim /etc/ansible/ansible.cfg
    [privilege_escalation]
      become=True
      become_method=sudo
      become_user=root
      become_ask_pass=True
  host Inventory配置文件默认在/etc/ansible/hosts 
    [web]
    172.16.62.151
    [db]
    172.16.62.152
    [all:children]
    web
    db
    

3.
  ansible all -m ping
  ansible all -a "/bin/echo hello"
  ansible all -m setup
  ansible all -m copy -a "src=/etc/hosts dest=/tmp/hosts"

  #state：present表示安装，latest表示升级最新版，absent表示删除    
  ansible db -m cron -a 'state=present minute="*/10" job="/usr/bin/echo hello" name="test cron job"'
  ansible db -a 'crontab -l'                   
  ansible db -a 'cat /var/spool/cron/crontabs/guo'  --sudo --ask-sudo-pass   

  ansible web -m apt -a "name=nginx state=present" --sudo --ask-sudo-pass
  ansible web -m service -a "name=nginx state=started" --sudo --ask-sudo-pass
  ansible web -m service -a "name=nginx state=stopped" -become
  ansible web -m apt -a "name=nginx state=absent" -become
 
 
4.
	Playbook命令集
	vim playbook.yml

	- hosts: all
	  become: yes
	  become_method: sudo
	  tasks:
	    - name: apt update
	      apt:
	        update_cache: yes
	    - name: Install apache
	      apt:
	        name: apache2
	        state: present
	      environment:
	        RUNLEVEL: 1  
	    - name: start apache
	      service:
	        name: apache2
	        state: started
	      environment:
	        RUNLEVEL: 1      
	    - name: Remove useless packages from the cache
	      apt:
	        autoclean: yes
	    - name: Remove dependencies that are no longer required
	      apt:
	        autoremove: yes  

ansible-playbook -i hosts playbook.yml
