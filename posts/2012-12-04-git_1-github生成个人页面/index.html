在github 创建个人网页

mkdir flyhawk007.github.com

cd flyhawk007.github.com

git init

touch README

git add README

git remote add origin git@github.com:flyhawk007/flyhawk007.github.com.git

echo “hello ,index.html” >index.html

git add .

git ci

git push -u origin master
