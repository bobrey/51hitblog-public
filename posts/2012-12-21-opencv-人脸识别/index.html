opencv 人脸识别

Fri Dec 21, 2012

300 Words|Read in about 1 Min
Tags: opencv   python  
  1. 正确安装Opencv2.3的python包,这个版本的包和之前不同，需要事先安装numpy，否则无法正常使用
  2. 将opencv安装目录里的\data\haarcascades\文件夹里的haarcascade_frontalface_alt.xml拷贝至本程序文件的所在目录，这是个训练样本文件，也可以从这里下载
  3. 电脑摄像头必须打开。 # -- coding: utf-8 -- “使用OpenCV 2.3进行实时人脸检测,按ESC退出” import cv2 import os import sys

pth = os.path.dirname(sys.argv[0]) cv2.namedWindow(“camera”) det = cv2.CascadeClassifier(pth +
“/haarcascade_frontalface_alt.xml”) cam = cv2.VideoCapture(0) while cam.isOpened(): flag, frame = cam.read(); if flag == True: tmp = cv2.cvtColor(frame, cv2.cv.CV_RGB2GRAY) objs = det.detectMultiScale(tmp, 1.1, 4, 0,
(frame.shape[1] / 7, frame.shape[0] / 7)) for rct in objs: cv2.rectangle(frame, (rct[0], rct[1]),
(rct[0] + rct[2], rct[1] + rct[3]),
(0, 0, 255), 2, cv2.CV_AA) cv2.imshow(“camera”, frame) if cv2.waitKey(10) == 27: cam.release() break

cv2.destroyWindow(“camera”)
