– (void)touchesBegan:(NSSet *)touches

withEvent:(UIEvent *)event{

UITouch *t = [touches anyObject];

CGPoint p = [t locationInView:self];

starLayer.position=p;

NSLog(@”»>touchesBegan»@%”,p);

}
